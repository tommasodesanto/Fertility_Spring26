% ==================== A SIMPLE MODEL ====================
\section{A Simple Model}

% 5
\begin{frame}{A Simple Model: Demographics and Timing}
  A stationary two-period overlapping-generations economy. A potential young household enters with a type
  \[
  i=(y_i,a_i),
  \]
  disposable income $y_i$ and entry wealth $a_i$ inherited from the old who exit as she arrives.
  \begin{itemize}
    \item Young households choose whether to enter the market, then tenure, occupied space, number of children, and saving.
    \item A young owner carries her house into old age and becomes next period's old incumbent; a young renter reaches old age with no house.
    \item Old households choose consumption, how much housing to keep, and the estate they leave, then exit.
  \end{itemize}
  Entry is a discrete choice against an outside value $\bar W^E$ with taste shocks of scale $\kappa_E$, so who reaches owner housing is set at entry. Births plus a renewal inflow form the next young cohort, and the exiting estates fund its entry wealth.
\end{frame}

% 6
\begin{frame}{Preferences}
  Children compete with their parents for the same budget and the same house: each child claims $\chi$ units of consumption and $\kappa$ units of space. Over consumption $c_i$, occupied housing $h_i$, and number of children $n_i$, the young household values
  \[
  u_i^Y(c_i,h_i,n_i)=\log(c_i-\chi n_i)+\alpha\log(h_i-\kappa n_i)+\vartheta\log n_i,
  \qquad \alpha,\vartheta>0.
  \]
  One more child means less space and less consumption left for the adults.

  Old households value consumption, retained housing, and a warm-glow estate $e_j$,
  \[
  u^O(c_j^O,h_j^O,e_j)=\log c_j^O+\gamma\log h_j^O+\mathcal B(e_j),
  \]
  and a young household discounts her own old age by $\beta$. Saving carries $a_i'\ge0$ into old age at gross return $1+r$; the savings load $k\equiv\beta(1+\gamma+b)$ is old age's claim on each unit of adult consumption.
\end{frame}

% 7
\begin{frame}{Markets and Technology}
  One fixed stock $\bar H$ of divisible floorspace, occupied by renting ($R$) or owning ($O$). The segments differ only in the largest size available,
  \[
  \HH^R=\{0<h\le\bar h^R\},\qquad \HH^O=\{0<h\le\bar h^O\},\qquad \bar h^R<\bar h^O,
  \]
  so family-sized space must be bought, not rented. All space clears in one market, and a no-arbitrage user cost ties the rental rate to the price:
  \[
  q=\rho P,\qquad \rho=r+\delta+\tau^p.
  \]
  The renter pays $q$ as rent; the owner pays it as interest, tax, depreciation, and forgone return --- equal per-unit value by arbitrage. Selling a house triggers a capital-gains tax that death forgives through basis step-up, a per-unit wedge on housing sold while alive,
  \[
  \bar\ell=\tau^g\,\frac{P\,\varpi}{1+\varpi},
  \]
  the only place the model touches inflation.
\end{frame}

% 8
\begin{frame}{The Household Problem, Mode by Mode}
  Conditional on entry, the household picks a tenure mode $m\in\{R,O\}$ and solves
  \[
  W_i^m=\max_{c_i,h_i,n_i,a_i'}\ u_i^Y(c_i,h_i,n_i)+\beta\big[\1\{m=R\}V^R(a_{j'})+\1\{m=O\}V^O(a_{j'},h_i)\big]
  \]
  subject to one budget and, for buyers, two financing tests:
  \[
  c_i+q h_i+a_i'=y_i+a_i,\qquad
  \1\{m=O\}\,(1-\phi)P h_i\le a_i,\qquad a_i'\ge0.
  \]
  Space costs $q$ per unit either way, so tenure bites only through the size limit and the down payment $(1-\phi)P h_i$, which wealth in hand must cover. An old incumbent nets $q-\bar\ell$ on each unit she sells, valuing kept space \emph{below} market by the tax she avoids; an old renter's demand is wedge-free.
\end{frame}

% 9
\begin{frame}{Stationary Competitive Equilibrium}
  A stationary equilibrium is a user cost $q$, an asset price $P=q/\rho$, young and old allocations, entry probabilities, and stationary measures such that:
  \begin{itemize}
    \item young households solve their mode problems and choose tenure $o_i=\arg\max\{W_i^R,W_i^O\}$; old households and entry are optimal;
    \item the single floorspace market clears,
    \[
    H^D(q)=\bar H;
    \]
    \item the old-incumbent measure, the young cohort, and the cohort mass reproduce themselves, and fiscal and housing-income accounts close by lump-sum transfers.
  \end{itemize}
  One price equates total demand to the fixed stock. Nothing in that condition asks whether the marginal family-sized unit is held by the household that values it most --- that question is the efficiency result below.
\end{frame}

% 10
\begin{frame}{All Frictions Collapse Into One Cap}
  Using $P=q/\rho$, owner housing is bounded by the tighter of the size limit and down-payment capacity:
  \[
  h_i\le H_i^O(q,\tau^p)=\min\left\{\bar h^O,\ \frac{a_i\rho}{(1-\phi)q}\right\},
  \qquad H_i^R=\bar h^R.
  \]
  Renting, the owner size limit, and finance all reduce to one effective family-housing cap. Substituting it into the fertility first-order condition prices the marginal child:
  \[
  \underbrace{\frac{\vartheta\,(c_i^m-\chi n_i^m)}{n_i^m}}_{\text{willingness to pay for a child}}
  =
  \underbrace{\chi+\kappa(q^m+\zeta_i^m)}_{\text{full price of a child}} .
  \]
  A binding cap acts as a per-child tax of $\kappa\zeta_i^m$: children cost more exactly where family-sized space is hard to reach, with market prices unchanged.
\end{frame}

% 11
\begin{frame}{The Access Gap}
  The access gap $\zeta_i^m$ is the goods value of one more unit of constrained family space, read off the household's own bundle. For a renter at the size limit,
  \[
  \frac{u^Y_{h,i}}{u^Y_{c,i}}=\frac{\alpha\,(c_i^R-\chi n_i^R)}{h_i^R-\kappa n_i^R}=q+\zeta_i^R,
  \]
  so she values space above the rent it commands. For an owner the gap separates finance from the size limit,
  \[
  \zeta_i^O=\underbrace{\frac{\mu_i}{\lambda_i^O}(1-\phi)P}_{\zeta_i^{O,F}\text{, down-payment margin}}+\underbrace{\frac{\theta_i^O}{\lambda_i^O}}_{\text{size-limit margin}}\ \ge0.
  \]
  Either way $\zeta_i$ is computable from observed consumption and housing, net of child needs: the young gap from bundles, the old wedge from deeds and the tax schedule.
\end{frame}

% 12
\begin{frame}{Where the Cap Binds, Fertility Responds}
  Under equal scaling at lifetime prices, $\chi=(1+k)\kappa q^m/\alpha$, the fertility response to relaxing the cap is proportional to the same gap that measures access:
  \[
  \Psi_i^m=\frac{\kappa}{\alpha}\,\frac{\zeta_i^m}{c_i^m-\chi n_i^m}\,\Omega_i^m,
  \qquad
  \Omega_i^m=\frac{\dfrac{\alpha}{h_i^m-\kappa n_i^m}+\dfrac{q^m}{c_i^m-\chi n_i^m}}{\dfrac{\vartheta}{(n_i^m)^2}+\dfrac{\chi^2}{(1+k)(c_i^m-\chi n_i^m)^2}+\dfrac{\alpha\kappa^2}{(h_i^m-\kappa n_i^m)^2}}>0.
  \]
  Holding exposure and pass-through fixed, a larger normalized gap means a larger birth response. The households closest to the cap are the ones whose fertility moves.
\end{frame}

% 13
\begin{frame}{Why the Stock Is Misallocated}
  Move one unit from an old incumbent to a financially constrained young owner:
  \[
  \underbrace{\left(q^O+\zeta_i^{O,F}\right)}_{\text{buyer's marginal value}}
  -\underbrace{\left(q-\bar\ell\right)}_{\text{incumbent's marginal value}}
  =
  \zeta_i^{O,F}+\frac{\bar\ell}{1+r}+\bar\ell>0 .
  \]
  The market price cancels. What is left is always positive: the young buyer's financing gap, plus her own anticipated capital-gains tax, plus the incumbent's realized one. The stock clears and is still misallocated --- not because the old occupy housing, but because their tax lock-in sits next to young households with unmet access gaps. The tax wedges are transfers, not resources, so the reallocation is a Pareto improvement.
\end{frame}

\section{The Quantitative Model}

% ---- The Quantitative Model divider ----
\begin{frame}
  \vfill
  \begin{center}
    \Large\textbf{The Quantitative Model}
  \end{center}
  \vfill
\end{frame}

% ---- Environment ----
\begin{frame}{Environment}
  A household lives $17$ four-year periods, ages $18$ to $82$; it enters at $18$ as a renter and dies at $82$. Write the child block as $(n,s)$: $n$ is the number of children and $s$ the child-aging stage.
  \begin{itemize}
    \item \textbf{Income:} a five-state persistent process $z\in\{0.6,0.8,1.0,1.2,1.4\}$, with stationary weights $(0.10,0.20,0.40,0.20,0.10)$ and Markov persistence $0.85$ --- and a deterministic age--earnings profile $e_j$.
    \item \textbf{Children:} up to two children per household. A birth opens one dependent-child stage that exits with hazard $2/9$ each period; when it exits the children leave home and their goods and space costs switch off.
    \item \textbf{Fertility:} completed fertility is chosen once, inside the fertile window ages $18$ to $42$, subject to taste shocks --- one shot, not a sequential birth hazard.
    \item \textbf{Choices:} the household picks \textbf{fertility} $\nu\in\{0,1,2\}$ while fertile, then \textbf{tenure} (rent or own), \textbf{size} (a room on the renter or owner menu), and \textbf{saving} $b'$.
  \end{itemize}
  So the two-period child cost $(\chi,\kappa)$ becomes a lifecycle of family-size-dependent needs that arrive with a birth and expire when children mature.
\end{frame}

% ---- Preferences ----
\begin{frame}{Preferences}
  {\small $n$: number of children; $s$: child-aging stage (stochastic); $m(n,s)=n\,\1\{1\le s\le K_c\}$: children currently at home, $K_c=1$.}

  \textbf{Flow utility.} Over residual consumption and residual housing services $x$:
  \[
  u(c,x;n,s)=\frac{\big[(c-\bar c(n,s))^{\alpha}\,x^{\,1-\alpha}\big]^{1-\sigma}}{1-\sigma}+\psi_{\mathrm{child}}\,m(n,s),
  \qquad \sigma=2.
  \]

  \textbf{Subsistence requirements.} Depend on family size:
  \[
  \bar c(n,s)=\bar c_0+\bar c_n\,m(n,s),\qquad
  \bar h(n,s)=\bar h_0+\underbrace{\bar h_{\mathrm{jump}}\,\1\{1\le s\le K_c\}}_{\text{first-child housing margin}}+\bar h_n\,m(n,s).
  \]

  \textbf{Housing services.} Renters take $x^R$ capped at $6$ rooms; owners get a service premium $\chi_o>1$:
  \[
  \underbrace{x^R=\max\{h^R-\bar h(n,s),\varepsilon\}}_{h^R\le h^R_{\max}=6},
  \qquad
  \underbrace{x^O=\chi_o\max\{h-\bar h(n,s),\varepsilon\}}_{\chi_o>1}.
  \]
  The additive child term is tied to children at home, so it shuts off once the child state matures.
\end{frame}

% ---- Bequests ----
\begin{frame}{Bequests}
  Terminal utility at death, over completed children $N(n,s)$:
  \[
  B(b,h,n,s)=\theta_0\max\{1+\theta_n N(n,s),0\}\,\frac{(\theta_1+\tilde b)^{1-\sigma}}{1-\sigma},
  \qquad
  \tilde b=\max\{b+\1_{\mathrm{own}}\,p\,h,0\}.
  \]
  \begin{itemize}
    \item $\theta_0$ governs the overall bequest motive (the saving level).
    \item $\theta_n$ scales the bequest with completed children $N(n,s)$ --- the parent--childless saving differential.
    \item $\theta_1$ is a shift on bequeathable wealth inside the CRRA term.
  \end{itemize}
  Bequeathable wealth $\tilde b$ uses the \emph{gross} owner housing value $p\,h$; the transaction-net sale value $(1-\psi)p\,h$ enters tenure transitions, not this terminal call.
\end{frame}

% ---- Housing, Tenure, and Budget Constraints ----
\begin{frame}{Housing, Tenure, and Budget Constraints}
  {\small Household chooses a room on the renter or owner menu: $h^R\le h^R_{\max}=6$; $h\in\{2,4,6,8,10\}$ (owner ladder). Family-sized rooms are reachable only through owning.}

  \textbf{Renters ($h=0$).} With rent $r(p)=(q+\delta+\tau_H)p$ and gross return $R=1+q$:
  \[
  b'=Rb+y(j,z)-c-r(p)\,h^R,\qquad b'\ge 0,\qquad h^R\le h^R_{\max}.
  \]

  \textbf{Owners.} Depreciation and property tax are flow costs of the occupied rung; borrowing is collateralized:
  \[
  b'=Rb+y(j,z)-c-(\delta+\tau_H)\,p\,h,\qquad b'\ge -\phi\,p\,h,\qquad \phi=0.80.
  \]
  \textbf{Tenure transitions.} Purchases subtract the full $p\,h'$; sales credit $(1-\psi)p\,h$; a purchase is feasible only if wealth in hand clears the down payment:
  \[
  b^+=
  \begin{cases}
  b-p\,h', & R\to O(h'),\\
  \max\{b+(1-\psi)p\,h,0\}, & O(h)\to R,\\
  b+(1-\psi)p\,h-p\,h', & O(h)\to O(h'),
  \end{cases}
  \qquad
  b+\1_{\mathrm{own}}(1-\psi)p\,h\ge (1-\phi)p\,h'.
  \]
  \begin{itemize}
    \item $q$: gross bond rate; $\delta$: depreciation; $\tau_H$: property tax; $\psi$: transaction cost; $\phi=0.80$: financed share.
  \end{itemize}
  Movers and newly matured children arrive as renters, then choose tenure by \mbox{argmax}.\footnote{Tenure is a deterministic $\arg\max$; the logit smoothing scale $\kappa_\tau\approx 8.3\times10^{-7}$ is calibrated to zero.}
\end{frame}

% ---- Births and Equilibrium ----
\begin{frame}{Births and Equilibrium}
  \textbf{Birth choice.} While fertile the household draws $\nu\in\{0,1,2\}$ from a logit over continuation values, with scale $\kappa_{\mathrm{fert}}$:
  \[
  \Pr(\nu\mid b,\tau,j,z)=\frac{\exp\{V_{\nu,s_\nu}(b,\tau,j,z)/\kappa_{\mathrm{fert}}\}}{\sum_{\ell=0}^{2}\exp\{V_{\ell,s_\ell}(b,\tau,j,z)/\kappa_{\mathrm{fert}}\}},
  \qquad
  \mathrm{TFR}=2\sum_{\nu=0}^{2}\nu\,\Omega_\nu.
  \]
  The calibrated scale $\kappa_{\mathrm{fert}}$ sits at its \emph{lower} bound: the calibration wants birth choices as sharp as allowed --- near-deterministic --- and the poorest types' birth probability is exactly zero. TFR doubles the per-household expected births because each fertility draw is one adult's contribution.

  \textbf{Market clearing.} One price $p$ equates housing demand to a unit-elastic supply curve:
  \[
  H^s(p)=H_0\left(\frac{(q+\delta+\tau_H)\,p}{\bar r}\right)^{\xi},\qquad \xi=1,\qquad H^d(p)=H^s(p).
  \]
  The theory section's fixed stock $\bar H$ is the limiting simplification of this ($\xi\to\infty$); the quantitative model clears against the coded unit-elastic supply.
\end{frame}

% ---- What Maps to What ----
\begin{frame}{What Maps to What}
  Every theory object has a quantitative counterpart --- these set up the bridge slides.
  \begin{center}
  \renewcommand{\arraystretch}{1.25}
  \begin{tabularx}{0.92\textwidth}{>{\raggedright\arraybackslash}p{0.40\textwidth}>{\raggedright\arraybackslash}X}
  \toprule
  simple model & quantitative model\\
  \midrule
  user cost $q=\rho P$ & user-cost flow $(\delta+\tau_H)p$ at the single price $p$\\
  down-payment friction $(1-\phi)P$ & collateral bound $b'\ge-\phi p h'$, $\phi=0.80$\\
  effective cap $\min\{\bar h^O,\,a_i\rho/((1-\phi)q)\}$ & owner ladder $\{2,4,6,8,10\}$ vs.\ renter cap $6$, against the down payment\\
  child costs $\chi,\ \kappa$ & consumption floor $\bar c(n,s)$ and housing floor $\bar h(n,s)$\\
  access gap $\zeta_i$ & shadow value of the collateral/size constraint on family space\\
  capital-gains wedge $\bar\ell$ & transaction cost $\psi$ on owner sales\\
  fixed stock $\bar H$ & unit-elastic supply $H^s(p)=H_0(r(p)/\bar r)^{\xi}$, $\xi=1$\\
  \bottomrule
  \end{tabularx}
  \end{center}
  The down payment is the load-bearing bridge: $\zeta_i^{O,F}$ in the theory and the $\phi=0.80$ collateral bound in the quantitative model. The one place the two differ openly is supply: the theory fixes the stock, the quantitative model clears a unit-elastic supply curve.
\end{frame}

% ==================== CALIBRATION AND FIT ====================
\section{Calibration and Fit}

% 13a
\begin{frame}{Calibration}
  \begin{itemize}
    \item The fit is a $14$-moment SMM table against $13$ estimated parameters.
    \item Empirical inputs: room and ownership moments from the ACS, and lifecycle wealth, tenure, and birth-housing moments from the PSID; entry wealth is the PSID young-childless-renter distribution, carried in rather than estimated.
    \item Externally fixed: the financed share $\phi=0.80$ and $\sigma=2$.
    \item The rental cap is not estimated. We set $\bar h^R=6$ rooms from an external rule --- the $90$th percentile of rental unit sizes, following Greaney, Parkhomenko, and Van Nieuwerburgh (2025), measured in our own ACS sample.\footnote{Search at $N_b=120$ wealth nodes and verify at $N_b=240$; a coarser grid tunes to numerical noise, so the reported fit is checked at the finer resolution.}
  \end{itemize}
  The precise statement is that the p90 of renter sizes falls in the $6$-room bin, so the maintained cap is a measured feature of the rental stock, not a free parameter.
\end{frame}

% 13b
\begin{frame}{The Data Do Not Fight the Cap}
  Re-solving at nearby cap values traces out a shallow valley around the external rule:
  \[
  \begin{array}{ccc}
  \toprule
  \bar h^R & \text{best loss} & \text{status}\\
  \midrule
  5.5 & 7.69 & \text{converged}\\

% 19
\begin{frame}{A Price Instrument Capitalizes but Barely Moves Births}
  \begin{itemize}
    \item Consider a $2\%$ tax on large homes.
    \item Large-home prices fall by $12.3\%$.
    \item Fertility hardly moves: $\Delta \mathrm{TFR}=+0.004$.
    \item The response is robust to numerical resolution, to the housing menu, and to the weighting of targets.
  \end{itemize}
  A pure price instrument capitalizes into house prices; it does not create births.
\end{frame}

% 20
\begin{frame}{Birth-Linked Purchase Grants Move Fertility}
  A grant paid at a birth-linked purchase raises fertility, and the response grows with the size of the grant:
  \[
  \begin{array}{cc}
  \toprule
  \text{grant size} & \Delta \mathrm{TFR}\\
  \midrule
  0.1 & +0.01\\
  0.4 & +0.13\\
  0.58 & +0.26\\
  1.0 & +0.49\\
  \bottomrule
  \end{array}
  \]
  Prices barely move --- large-home prices rise about $1\%$.
  \begin{itemize}
    \item One caveat: a small conditional grant can pull purchases back before the birth, delaying them through anticipation.
  \end{itemize}
  Grants move the birth margin because they relax the owner financing gap $\zeta_i^{O,F}$.
\end{frame}

% 21
\begin{frame}{A Package That Decentralizes the Reallocation}
  Tax the incumbent stock and spend the revenue on young buyers' access:
  \[
  \text{large-home tax}+\text{birth-linked purchase grant}
  \quad\Rightarrow\quad
  \Delta \mathrm{TFR}=+0.15,\quad
  \Delta p_{\text{large}}=-11\%.
  \]
  The tax lowers what incumbents will pay to keep large units; the grant closes the young side's financing gap at the birth margin. Revenue-funded, it is the decentralized version of the planner's move --- and it works through access, not price.
\end{frame}

% 22
\begin{frame}{The Bridge, in Three Numbers}
  \begin{enumerate}
    \item Access: fertile renters at the cap have $\zeta/q=1.28$, and $0.734$ of the fertility-response mass sits on households with $\zeta>0$.
    \item Allocation: $87\%$ of the family-sized stock is held by post-fertile households.
    \item Policy: the tax-funded access transfer raises TFR by $0.15$ while cutting large-home prices $11\%$.
  \end{enumerate}
  The same object $\zeta$ measures fertility exposure and stock misallocation, and the package is the decentralized version of the planner's reallocation.
\end{frame}

% 23
\begin{frame}{Where This Version Falls Short}
  \begin{itemize}
    \item Several parameters sit at their search bounds.\footnote{Eight of the $13$ estimated parameters are at or against a bound at the current calibration --- evidence that the parameter space is compensating for missing mechanisms, not a failure of the search.}
    \item The fertility taste-shock scale $\kappa_{\mathrm{fert}}$ is at its \emph{lower} bound: birth choice is near-deterministic and the poorest households' birth probability is exactly zero. This strengthens the reading that the child-price structure, not the taste shocks, drives the misses --- and a near-$\arg\max$ birth choice makes the fertility moments numerically sensitive to small changes in the price of family space.
    \item Fertility is mistimed: births cluster at $33$, against a data peak near $27$.
    \item Planned extension: a child time cost, so children are priced partly in foregone income.
    \item Planned extensions: an old-age exit margin, and childlessness targets consistent with the estimand.
  \end{itemize}
  These belong in the draft as honest limitations, not as settled calibration details.
\end{frame}
